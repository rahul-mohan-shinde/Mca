/*
for the static variables the JVM provide the default values
for the local variables the jvm won't provide the default 
1. instance variables
2. static varables
3. 1-instance methods
4. static method
*/

class Test1 {
	//instance variables
	int a = 10;
	boolean b;
	//static variables
	static int c = 20;
	static double d;
	//instance methods
	void m1(){
		System.out.println(a);
		System.out.println(b);
		System.out.println(c);
		System.out.println(d);
	}
	//static method
	static void m2(){
		Test1 t = new Test1();
		System.out.println(t.a);
	}
	public static void main(String[] args){
		Test1 t = new Test1();
		t.m1();
		m2();
	}
}

// to print a reverse of the number
class reverse{
	public static void main(String[] args){
		int n = 123;
		int rev = 0;
		while( n > 0){
			int r = n % 10;
			rev = rev * 10 + r;
 			n = n / 10;
		}
		System.out.println(rev);
	}
}

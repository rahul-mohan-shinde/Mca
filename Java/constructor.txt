  float width ;

    length = l;
    width = w;

  }

  rectangle()
  {
    System.out.println("Constructing Rectangle...");
    float length =0;
    float width =0;

  }
  rectangle(float val)
  {
    System.out.println("Constructing Rectangle...");
     length = width = val;
  }

  // compute and return area of a rectangle
  float area()

  }


}
